				SELECT 
                tick.tick_id,
                tick.tick_titulo,
                tick.tick_descrip,
                tick.tick_estado,
                tick.fech_crea,
                tick.fech_cierre,
                tick.fech_asig,
                IF(tick.tick_estre IS NULL,0,tick.tick_estre) as Estrellas,
                CONCAT(usu.usu_nom,' ',usu.usu_ape) AS Usuario,
                cat.cat_nom,
                cats.cats_nom,
                prio.prio_nom,
               	IF(asig.usu_nom IS NULL,'Pendiente',CONCAT(asig.usu_nom,' ',asig.usu_ape)) as Asignado,
                'Total' as Total
                FROM 
                tm_ticket tick
                INNER join tm_categoria cat on tick.cat_id = cat.cat_id
                INNER join tm_subcategoria cats on tick.cats_id = cats.cats_id
                INNER join tm_usuario usu on tick.usu_id = usu.usu_id
                LEFT join tm_usuario asig on tick.usu_asig = asig.usu_id
                INNER join tm_prioridad prio on tick.prio_id = prio.prio_id
                WHERE
                tick.est = 1
